  counter <- 0

    if (x[i] ==0)
    counter <- counter + 1
  }

multi_matrix <- function() {
  m <- matrix(nrow=3, ncol=3)
  for (i in 1:nrow(m)) {
    for (j in 1:ncol(m)){
      m[i,j] <- i * j
    }
  }
  
 return(m)
}
