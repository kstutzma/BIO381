  }

# FUNCTION: multi_matrix
# packages: none
# purpose: to multiply the two loops in a matrix
# input: empty matrix
# output: matrix filled with multiplied loops
# -------------------------------

multi_matrix <- function() {
  m <- matrix(nrow=3, ncol=3)
  for (i in 1:nrow(m)) {
    for (j in 1:ncol(m)){
      m[i,j] <- i * j
    }
  }
  
 return(m)
}

shuffle_df <- function(z=df_sim){
